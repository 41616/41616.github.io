按照左手数字排序肯定不对，又考虑到题目里只出现乘除法，就想到按照左右手乘积排序。

```cpp
#include<iostream>
#include<algorithm>
using namespace std;
typedef unsigned long long ull;
struct person{
	ull l,r;
	bool operator<(person&i)const{
		return l*r<i.l*i.r;
	}
	void InPut(){
		cin>>l>>r;
	}
}p[1024];
int main(){
	ull n,l,r,ans=0;cin>>n>>l>>r;
	for(ull i=0;i<n;i++)p[i].InPut();
	sort(p,p+n);
	for(ull i=0;i<n;i++){
		ans=max(ans,l/p[i].r);
		l*=p[i].l;
	}
	cout<<ans;
}
```
